function rootReducer(state=initialState, action) {
    switch(action.type) {
        case WITHDRAW:
            return {
                balance: state.balance - action.amount
            };
        default:
            return state;
            
        case DEPOSIT:
            return {
                balance: state.balance + action.amount
            };
        default:
            return state;
            
        case TRANSFERFUNDS:
            return {
                balance: state.balance - action.amount
            };
        default:
            return state;    
     }
}
